# For Example #

import numpy as np
import os
import importlib.util

os.add_dll_directory("C:\\Program Files\\Lumerical\\v202\\api\\python")

lumapi = importlib.util.module_from_spec(spec)
spec.loader.exec_module(lumapi)
fdtd = lumapi.FDTD()

MyMaterial = fdtd.addmaterial("Chi_Three_Full_Tensorial_ISO_V_1_0_0")
fdtd.setmaterial(MyMaterial,"name","Chi_Three_Full_Tensorial_ISO")
fdtd.setmaterial("Chi_Three_Full_Tensorial_ISO", "Anisotropy", "Diagonal")

Chi_1 = 2;
Chi_3 = 9e-16;

fdtd.setmaterial("Chi_Three_Full_Tensorial_ISO", "Chi1", Chi_1);
fdtd.setmaterial("Chi_Three_Full_Tensorial_ISO", "Chi3", Chi_3);
